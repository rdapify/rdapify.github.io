🎯 تأثير الأداء​

تجميع الاتصالات​

  • أسرع بنسبة 30-40% للاستعلامات المتكررة
  • حمل إضافي ضئيل للاتصالات الجديدة
  • تنظيف تلقائي للاتصالات الخاملة

جمع المقاييس​

  • حمل إضافي ~1-2ms لكل استعلام
  • حد افتراضي 10,000 مقياس
  • تنظيف تلقائي للبيانات القديمة

التسجيل​

  • تأثير صفري عند التعطيل
  • إخراج غير متزامن لا يحظر
  • إسهاب قابل للتكوين

📈 مثال الاستخدام​

import { RDAPClient } from 'rdapify';

const client = new RDAPClient({
  cache: true,
  logging: { level: 'info', enabled: true },
});

// تنفيذ الاستعلامات
await client.domain('example.com');
await client.ip('8.8.8.8');

// الحصول على المقاييس
const metrics = client.getMetrics();
console.log(`النجاح: ${metrics.successRate}%`);
console.log(`متوسط الوقت: ${metrics.avgResponseTime}ms`);
console.log(`نجاح ذاكرة التخزين المؤقت: ${metrics.cacheHitRate}%`);

// الحصول على إحصائيات التجميع
const pool = client.getConnectionPoolStats();
console.log(`الاتصالات: ${pool.activeConnections}/${pool.totalConnections}`);

// الحصول على السجلات
const logs = client.getLogs(5);
logs.forEach(log => console.log(`[${log.level}] ${log.message}`));

// التنظيف
client.destroy();
